class Solution {
    public ListNode mergeNodes(ListNode head) {
        ListNode temp= head;
        ListNode result= new ListNode(0);
        ListNode current = result;

        while(temp!=null ){
            int s=0;
            while (temp.val!=0){
                s+=temp.val;
                temp=temp.next;
            }
            current.next= new ListNode(s);
            current= current.next;
            temp=temp.next;
        }
        return result.next.next;
    }
}
